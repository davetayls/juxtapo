200 	 * Listener attached to the keydown event on the body
201 	 * @private

203     juxtapo.onBody_KeyDown = function(e){
204         var keycode;
205         if (window.event) 
206             keycode = window.event.keyCode;
207         else 
208             if (e) 
209                 keycode = e.which;
210             else 
211                 return true;
212         
213         juxtapo.eh.logInfo("keycode is: " + keycode); // ##DEBUG
214         // Check if user presses Ctl+o or Ctl+right
215         if (e.ctrlKey && (keycode == 79 || keycode == 39)) {
216             juxtapo.templates.forward();
217             juxtapo.utils.preventDefaultEventAction(e);
218             return false;
219         }
220         // Check if user presses Ctl+u or Ctl+left
221         if (e.ctrlKey && (keycode == 85 || keycode == 37)) {
222             juxtapo.templates.back();
223             juxtapo.utils.preventDefaultEventAction(e);
224             return false;
225         }
226         // Check if user presses Ctl+up-arrow
227         if (e.ctrlKey && keycode == 38) {
228             juxtapo.templates.change(true);
229             juxtapo.utils.preventDefaultEventAction(e);
230             return false;
231         }
232         // Check if user presses Ctl+down-arrow
233         if (e.ctrlKey && keycode == 40) {
234             juxtapo.templates.change(false);
235             juxtapo.utils.preventDefaultEventAction(e);
236             return false;
237         }
238         // Ctl+Space
239         if (e.ctrlKey && keycode == 32) {
240             juxtapo.control.toggle();
241             juxtapo.utils.preventDefaultEventAction(e);
242             return false;
243         }
244         // nudge designs
245         if (e.ctrlKey && keycode == 73) {
246             var pixels = e.shiftKey ? 1 : 25;
247             juxtapo.templates.nudge("top", pixels);
248             juxtapo.utils.preventDefaultEventAction(e);
249             return false;
250         }
251         if (e.ctrlKey && keycode == 76) {
252             var pixels = e.shiftKey ? 1 : 25;
253             juxtapo.templates.nudge("right", pixels);
254             juxtapo.utils.preventDefaultEventAction(e);
255             return false;
256         }
257         if (e.ctrlKey && keycode == 75) {
258             var pixels = e.shiftKey ? 1 : 25;
259             juxtapo.templates.nudge("bottom", pixels);
260             juxtapo.utils.preventDefaultEventAction(e);
261             return false;
262         }
263         if (e.ctrlKey && keycode == 74) {
264             var pixels = e.shiftKey ? 1 : 25;
265             juxtapo.templates.nudge("left", pixels);
266             juxtapo.utils.preventDefaultEventAction(e);
267             return false;
268         }
269         
270         return true;
271     };
